#subplot() function
import numpy as np
from matplotlib import pyplot as plt
x=np.arange(1,10,0.1)
y_cos=np.cos(x)
y_sin=np.sin(x)
#setting sub plots
plt.subplot(3,1,1)
#first plot
plt.plot(x,y_sin)
plt.title("Sine curve")
#second plot
plt.subplot(3,1,3)
plt.plot(x,y_cos)
plt.title("Cosine curve")

plt.show()

from matplotlib import pyplot as plt

x = range(N)
